CREATE DATABASE IF NOT EXISTS USERS; 

USE USERS;

CREATE TABLE USERS_INFO(
	
ID INT PRIMARY KEY AUTO_INCREMENT,
	
USUARIO VARCHAR(150),
    
NOMBRE_APELLIDO VARCHAR(150),
    
CORREO VARCHAR(150),
    
TELEFONO VARCHAR(150),
    
DIRECCION VARCHAR(250),
    
ADMIN BOOL,
    
PASSWORD VARCHAR(100)
);
